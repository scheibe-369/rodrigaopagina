          CUBO: { id: 'cubo', name: 'Cubo Self', theme: 'from-cyan-500 to-blue-600', text: 'text-cyan-400', bg: 'bg-cyan-500/10', border: 'border-cyan-500/30' },

          { id: '6', title: 'Ajustar Lances de Leilão de Busca', client: CLIENTES.CUBO, startDate: '2026-04-08', deadline: 'Hoje', status: 'doing', progress: 40, platform: 'Google Ads', notes: '' },

          { id: '8', title: 'Relatório Mensal de Março', client: CLIENTES.VIV, startDate: '2026-04-01', deadline: '05/04', status: 'done', progress: 100, platform: 'Data Studio' },

          { id: '1', title: 'CRIATIVO NOVO: Carrossel Dia das Mães', client: CLIENTES.VIV, content: 'Precisamos de 3 imagens no estilo minimalista mostrando a nova coleção. Focar nas cores pastéis.', link: 'https://drive.google.com/exemplo-referencia', status: 'pendente' }
        ];

        const ANUNCIOS_MOCK = [
          { id: '101', client: 'moteis', platform: 'Meta', title: 'Carrossel Suítes Premium', copy: 'Surpreenda quem você ama neste final de semana. Conheça novas suítes premium com hidro e teto solar.', imageUrl: 'https://images.unsplash.com/photo-1582719478250-c89cae4dc85b?auto=format&fit=crop&q=80&w=400&h=300' },
          { id: '202', client: 'cubo', platform: 'Meta', title: 'Vídeo Institucional - Segurança', copy: 'Seus bens guardados com a mesma segurança de um banco. 🔒 Assista ao vídeo.', imageUrl: 'https://images.unsplash.com/photo-1600585154340-be6161a56a0c?auto=format&fit=crop&q=80&w=400&h=300' },
        ];

        // Auxiliares de LocalStorage
        const loadLocal = (key, fallback) => {
            try {
                const item = window.localStorage.getItem('trafego_' + key);
                return item ? JSON.parse(item) : fallback;
            } catch (error) {
                return fallback;
            }
        };

        const saveLocal = (key, data) => {
            try {
                window.localStorage.setItem('trafego_' + key, JSON.stringify(data));
            } catch (error) {
                console.error("Erro ao salvar localmente", error);
            }
        };

        function App() {
          const [user, setUser] = useState(null);
          const [db, setDb] = useState(null);
          const [appId, setAppId] = useState('trafego-app-local');

          const [activeAdTab, setActiveAdTab] = useState('moteis');
          
          // Inicia lendo do LocalStorage (ou usando Mock se estiver vazio)
          const [tasks, setTasks] = useState(() => loadLocal('tasks', TAREFAS_MOCK));
          const [anuncios, setAnuncios] = useState(() => loadLocal('anuncios', ANUNCIOS_MOCK));
          const [criativos, setCriativos] = useState(() => loadLocal('criativos', CRIATIVOS_MOCK));

          const [isAdModalOpen, setIsAdModalOpen] = useState(false);
          const [isCreativeModalOpen, setIsCreativeModalOpen] = useState(false);
          const [isEditTaskModalOpen, setIsEditTaskModalOpen] = useState(false);
          const [editingTask, setEditingTask] = useState(null);

          const [newAd, setNewAd] = useState({ title: '', copy: '', client: 'moteis', platform: 'Meta', imageUrl: '' });
          const [newCreative, setNewCreative] = useState({ title: 'CRIATIVO NOVO: ', content: '', client: 'moteis', link: '' });

          // --- CONFIGURAÇÃO DO FIREBASE (Caso decida usar depois para sincronizar) ---
          useEffect(() => {
              const initFirebase = async (fb) => {
                  let config;
                  let currentAppId = 'trafego-app-local';
                  
                  if (typeof __firebase_config !== 'undefined') {
                      config = JSON.parse(__firebase_config);
                      currentAppId = typeof __app_id !== 'undefined' ? __app_id : currentAppId;
                  } else {
                      config = { apiKey: "COLE_SUA_API_KEY_AQUI", authDomain: "", projectId: "", appId: "" };
                      if(config.apiKey.includes("COLE_SUA")) return; // Continua no modo offline/localStorage
                  }
                  
                  const app = fb.initializeApp(config);
                  const auth = fb.getAuth(app);
                  const firestoreDb = fb.getFirestore(app);
                  setDb(firestoreDb);
                  setAppId(currentAppId);
                  
                  if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                      await fb.signInWithCustomToken(auth, __initial_auth_token);
                  } else {
                      await fb.signInAnonymously(auth);
                  }
                  fb.onAuthStateChanged(auth, setUser);
              };

              const checkFirebase = setInterval(() => {
                  if (window.FirebaseAPI) {
                      clearInterval(checkFirebase);
                      initFirebase(window.FirebaseAPI);
                  }
              }, 100);
              return () => clearInterval(checkFirebase);
          }, []);

          // --- SINCRONIZAÇÃO EM TEMPO REAL (FIREBASE) ---
          useEffect(() => {
              if (!user || !db || !window.FirebaseAPI) return;
              const { collection, onSnapshot } = window.FirebaseAPI;
              
              const listenToCollection = (colName, setter, mockData) => {
                  const colRef = collection(db, 'artifacts', appId, 'public', 'data', colName);
                  return onSnapshot(colRef, (snapshot) => {
                      const loaded = [];
                      snapshot.forEach(doc => loaded.push({ id: doc.id, ...doc.data() }));
                      if(loaded.length === 0) {
                          mockData.forEach(item => syncData(colName, item));
                      } else {
                          setter(loaded);
                      }
                  }, (err) => console.error(err));
              };

              const unsubTasks = listenToCollection('tasks', setTasks, loadLocal('tasks', TAREFAS_MOCK));
              const unsubAnuncios = listenToCollection('anuncios', setAnuncios, loadLocal('anuncios', ANUNCIOS_MOCK));
              const unsubCriativos = listenToCollection('criativos', setCriativos, loadLocal('criativos', CRIATIVOS_MOCK));
              
              return () => { unsubTasks(); unsubAnuncios(); unsubCriativos(); };
          }, [user, db, appId]);

          // FUNÇÕES DE SALVAR NO BANCO OU LOCALSTORAGE
          const syncData = async (colName, itemToSync) => {
              // 1. Atualiza o Estado Local (Tela)
              let newState = [];
              if(colName === 'tasks') {
                  setTasks(curr => { 
                      const i = curr.findIndex(x=>String(x.id)===String(itemToSync.id)); 
                      newState = i>-1 ? Object.assign([], curr, {[i]: itemToSync}) : [itemToSync, ...curr]; 
                      saveLocal('tasks', newState); // Salva no navegador
                      return newState; 
                  });
              }
              if(colName === 'anuncios') {
                  setAnuncios(curr => { 
                      const i = curr.findIndex(x=>String(x.id)===String(itemToSync.id)); 
                      newState = i>-1 ? Object.assign([], curr, {[i]: itemToSync}) : [itemToSync, ...curr]; 
                      saveLocal('anuncios', newState);
                      return newState; 
                  });
              }
              if(colName === 'criativos') {
                  setCriativos(curr => { 
                      const i = curr.findIndex(x=>String(x.id)===String(itemToSync.id)); 
                      newState = i>-1 ? Object.assign([], curr, {[i]: itemToSync}) : [itemToSync, ...curr]; 
                      saveLocal('criativos', newState);
                      return newState; 
                  });
              }

              // 2. Se Firebase estiver ativo, salva lá também
              if (user && db && window.FirebaseAPI) {
                  const { doc, setDoc } = window.FirebaseAPI;
                  const docRef = doc(db, 'artifacts', appId, 'public', 'data', colName, String(itemToSync.id));
                  await setDoc(docRef, itemToSync);
              }
          };

          const deleteData = async (colName, itemId) => {
              if(colName === 'criativos') {
                  setCriativos(curr => {
                      const newState = curr.filter(c => String(c.id) !== String(itemId));
                      saveLocal('criativos', newState);
                      return newState;
                  });
              }

              if (user && db && window.FirebaseAPI) {
                  const { doc, deleteDoc } = window.FirebaseAPI;
                  const docRef = doc(db, 'artifacts', appId, 'public', 'data', colName, String(itemId));
                  await deleteDoc(docRef);
              }
          };


          // --- LÓGICAS DA APLICAÇÃO ---
          const doingTasks = tasks.filter(t => t.status === 'doing');
          const queueTasks = tasks.filter(t => t.status !== 'doing' && t.status !== 'inbox' && t.status !== 'done');
          const inboxTasks = tasks.filter(t => t.status === 'inbox');
          const doneTasks = tasks.filter(t => t.status === 'done');

          const getTempoDecorrido = (dataStr) => {
            if (!dataStr) return '0 dias';
            const inicio = new Date(dataStr);
            const hoje = new Date();
            const diffTime = Math.abs(hoje - inicio);
            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
            return diffDays === 1 ? '1 dia' : `${diffDays} dias`;
          };

          const aprovarParaFila = (taskId) => {
            const task = tasks.find(t => String(t.id) === String(taskId));
            if(task) syncData('tasks', { ...task, status: 'todo' });
          };

          const novaSolicitacaoJuan = () => {
            const newTask = {
              id: Date.now().toString(),
              title: 'Nova Ideia / Demanda do Juan 💡',
              client: CLIENTES.CUBO,
              deadline: 'A definir',
              status: 'inbox',
              platform: 'Ideia'
            };
            syncData('tasks', newTask);
          };

          const handleAddAd = (e) => {
            e.preventDefault();
            syncData('anuncios', { ...newAd, id: Date.now().toString() });
            setIsAdModalOpen(false);
            setNewAd({ title: '', copy: '', client: 'moteis', platform: 'Meta', imageUrl: '' });
          };

          const handleAddCreative = (e) => {
            e.preventDefault();
            const clienteObj = Object.values(CLIENTES).find(c => c.id === newCreative.client);
            syncData('criativos', { ...newCreative, id: Date.now().toString(), client: clienteObj, status: 'pendente' });
            setIsCreativeModalOpen(false);
            setNewCreative({ title: 'CRIATIVO NOVO: ', content: '', client: 'moteis', link: '' });
          };

          const concluirCriativo = (id) => {
            deleteData('criativos', id);
          };

          const abrirEdicaoTarefa = (task) => {
            setEditingTask({ ...task });
            setIsEditTaskModalOpen(true);
          };

          const salvarEdicaoTarefa = (e) => {
            e.preventDefault();
            syncData('tasks', editingTask);
            setIsEditTaskModalOpen(false);
          };

          const moverParaFinalizado = (taskId) => {
            const task = tasks.find(t => String(t.id) === String(taskId)) || editingTask;
            if(task) syncData('tasks', { ...task, status: 'done', progress: 100 });
            if(isEditTaskModalOpen) setIsEditTaskModalOpen(false);
          };

          const atualizarNotaTarefa = (taskId, novaNota) => {
            setTasks(current => current.map(t => String(t.id) === String(taskId) ? { ...t, notes: novaNota } : t));
          };
          const salvarNotaFinal = (task) => {
            syncData('tasks', task);
          };

          return (
            <div className="min-h-screen bg-[#070b14] font-sans text-gray-200 overflow-x-hidden selection:bg-fuchsia-500/30 pb-20">
              
              <div className="fixed top-0 left-0 w-full h-full overflow-hidden -z-10 pointer-events-none">
                <div className="absolute top-[-10%] left-[-10%] w-[40%] h-[40%] rounded-full bg-purple-900/20 blur-[120px]"></div>
                <div className="absolute bottom-[-10%] right-[-10%] w-[40%] h-[40%] rounded-full bg-blue-900/20 blur-[120px]"></div>
              </div>

              <main className="max-w-7xl mx-auto p-6 lg:p-10 space-y-12">
                
                <header className="flex flex-col md:flex-row md:items-end justify-between gap-6 border-b border-white/5 pb-8">
                  <div>
                    <div className="flex items-center gap-2 mb-2">
                      <i className="fa-solid fa-wand-magic-sparkles text-fuchsia-400"></i>
                      <span className="text-sm font-semibold tracking-wider text-fuchsia-400 uppercase">Tráfego Studio</span>
                    </div>
                    <h1 className="text-4xl lg:text-5xl font-black text-transparent bg-clip-text bg-gradient-to-r from-white via-gray-200 to-gray-500 tracking-tight">
                      Central de Produção

                    <p className="text-gray-400 mt-2 flex items-center gap-2">
                      <span className="relative flex h-3 w-3">
                        <span className={`animate-ping absolute inline-flex h-full w-full rounded-full opacity-75 ${user ? 'bg-green-400' : 'bg-blue-400'}`}></span>
                        <span className={`relative inline-flex rounded-full h-3 w-3 ${user ? 'bg-green-500' : 'bg-blue-500'}`}></span>

                      <div className="text-2xl font-bold text-white">{doingTasks.length < 10 ? `0${doingTasks.length}` : doingTasks.length}</div>

                  <div className="flex flex-col md:flex-row justify-between items-start md:items-center gap-4 mb-6 relative z-10">
                    <div className="flex items-center gap-3">
                      <div className="p-3 bg-yellow-500/20 text-yellow-400 rounded-lg shadow-[0_0_15px_rgba(234,179,8,0.2)] flex items-center justify-center">
                        <i className="fa-solid fa-inbox"></i>
                      </div>
                      <div>
                        <h2 className="text-2xl font-bold text-white">Solicitações do Juan</h2>
                        <p className="text-sm text-gray-400">Aguardando aprovação e triagem do Marketing</p>
                      </div>
                    </div>
                    <button 
                      onClick={novaSolicitacaoJuan}

                    >
                      <i className="fa-solid fa-plus"></i> Nova Demanda
                    </button>
                  </div>

                    {inboxTasks.map(task => (
                      <div key={task.id} className="bg-gray-900/60 border border-yellow-500/20 p-5 rounded-2xl flex flex-col justify-between hover:border-yellow-500/40 transition-colors shadow-sm">
                        <div>
                          <div className="flex justify-between items-start mb-3">
                            <span className={`px-2 py-1 text-[10px] font-bold uppercase tracking-wider rounded-md ${task.client.bg} ${task.client.text} border ${task.client.border}`}>
                              {task.client.name}
                            </span>
                            <span className="text-[10px] font-bold uppercase tracking-wider text-yellow-500 bg-yellow-500/10 px-2 py-1 rounded-md border border-yellow-500/20 flex items-center gap-1">
                              <span className="w-1.5 h-1.5 bg-yellow-500 rounded-full animate-pulse"></span> Pendente
                            </span>
                          </div>
                          <h4 className="text-white font-semibold mb-2">{task.title}</h4>
                          <p className="text-xs text-gray-400">Prazo sugerido: <strong className="text-gray-300">{task.deadline}</strong></p>
                        </div>
                        
                        <button 
                          onClick={() => aprovarParaFila(task.id)}
                          className="mt-4 w-full flex items-center justify-center gap-2 bg-white/5 hover:bg-green-500/20 text-gray-300 hover:text-green-400 border border-white/10 hover:border-green-500/30 py-2.5 rounded-xl text-xs font-bold transition-all group"
                        >
                          <i className="fa-solid fa-check group-hover:scale-125 transition-transform"></i> Aprovar e Mover p/ Fila
                        </button>
                      </div>
                    ))}
                    
                    {inboxTasks.length === 0 && (
                      <div className="col-span-full py-8 text-center border border-white/5 border-dashed rounded-2xl bg-white/[0.01]">
                        <p className="text-gray-500">Nenhuma solicitação pendente. Caixa de entrada zerada! 🎉</p>
                      </div>
                    )}
                  </div>
                </section>

                <div className="grid grid-cols-1 lg:grid-cols-12 gap-8">
                  
                  <section className="lg:col-span-7 space-y-6">
                    <div className="flex items-center gap-3">
                      <div className="p-3 bg-orange-500/20 text-orange-400 rounded-lg flex items-center justify-center">
                        <i className="fa-solid fa-fire"></i>
                      </div>
                      <h2 className="text-2xl font-bold text-white">O que estamos fazendo (Tráfego)</h2>
                    </div>
                    
                    <div className="grid gap-5">
                      {doingTasks.map(task => (
                        <div key={task.id} className="relative group rounded-3xl bg-gray-900/50 border border-white/5 p-1 overflow-hidden backdrop-blur-sm hover:border-white/20 transition-all duration-500">
                          <div className={`absolute top-0 left-0 w-1 h-full bg-gradient-to-b ${task.client.theme}`}></div>
                          <div className="p-6">
                            <div className="flex justify-between items-start mb-4">
                              <span className={`px-3 py-1 text-xs font-bold uppercase tracking-wider rounded-full ${task.client.bg} ${task.client.text} border ${task.client.border}`}>
                                {task.client.name}
                              </span>
                              <div className="flex gap-2">
                                <span className="flex items-center gap-1.5 text-xs font-medium text-gray-400 bg-black/40 px-3 py-1 rounded-full">
                                  <i className="fa-regular fa-clock text-orange-400"></i> Entrega: {task.deadline}
                                </span>
                                <button onClick={() => abrirEdicaoTarefa(task)} className="bg-white/5 hover:bg-white/10 text-gray-300 px-3 py-1 rounded-full text-xs font-bold transition-colors border border-white/10">
                                  <i className="fa-solid fa-pen"></i> Editar
                                </button>
                              </div>
                            </div>
                            
                            <h3 className="text-xl font-bold text-white mb-4 pr-4">{task.title}</h3>
                            
                            <div className="flex items-center gap-4 mb-6 bg-black/30 p-3 rounded-xl border border-white/5">
                                <div className="flex flex-col">
                                    <span className="text-[10px] text-gray-500 uppercase font-bold tracking-wider mb-1">Iniciado em</span>
                                    <span className="text-sm font-semibold text-gray-300">{task.startDate ? new Date(task.startDate + 'T12:00:00').toLocaleDateString('pt-BR') : 'Não definido'}</span>
                                </div>
                                <div className="w-px h-8 bg-white/10"></div>
                                <div className="flex flex-col">
                                    <span className="text-[10px] text-gray-500 uppercase font-bold tracking-wider mb-1">Tempo na Esteira</span>
                                    <span className="text-sm font-bold text-orange-400">{getTempoDecorrido(task.startDate)}</span>
                                </div>
                            </div>
                            
                            <div className="space-y-2">
                              <div className="flex justify-between text-xs text-gray-400 font-medium">
                                <span>Progresso da Demanda</span>
                                <span className={task.client.text}>{task.progress}%</span>
                              </div>
                              <div className="w-full bg-black/50 rounded-full h-2.5 overflow-hidden">
                                <div 
                                  className={`h-2.5 rounded-full bg-gradient-to-r ${task.client.theme} relative`}
                                  style={{ width: `${task.progress}%` }}
                                >
                                  <div className="absolute top-0 right-0 bottom-0 w-4 bg-white/30 animate-pulse"></div>
                                </div>
                              </div>
                            </div>
                            
                            {/* EDITOR DE OBSERVAÇÕES COLORIDO */}
                            <div className={`mt-6 p-1 rounded-2xl bg-gradient-to-br ${task.client.theme} shadow-lg shadow-black/20`}>
                              <div className="bg-black/40 backdrop-blur-md rounded-xl p-3">
                                <div className="flex items-center justify-between mb-2">
                                  <span className="text-xs font-bold text-white uppercase tracking-wider flex items-center gap-2">
                                    <i className="fa-solid fa-pen-to-square"></i> Observações / Mural
                                  </span>
                                </div>
                                <textarea 
                                  className="w-full bg-black/20 text-white placeholder-white/50 text-sm p-3 rounded-lg border border-white/10 outline-none focus:border-white/40 focus:bg-black/40 transition-all resize-y min-h-[80px]"
                                  placeholder="Escreva aqui suas ideias, o que a equipe disse, links úteis..."
                                  value={task.notes || ''}
                                  onChange={(e) => atualizarNotaTarefa(task.id, e.target.value)}
                                  onBlur={() => salvarNotaFinal(task)}
                                />
                              </div>
                            </div>

                          </div>
                        </div>
                      ))}

                        {Object.values(CLIENTES).map(client => (

                            key={client.id}

                          </button>

                      </div>
                      
                      <button 
                        onClick={() => setIsAdModalOpen(true)}

                            />
                          </div>

                      <div className="grid grid-cols-2 gap-4">
                          <div>
                            <label className="block text-xs font-bold text-gray-400 uppercase tracking-wider mb-2">Progresso (%)</label>

                              className="w-full bg-black/50 border border-white/10 rounded-xl px-4 py-3 text-white outline-none focus:border-blue-500/50"

                          </div>

                            <label className="block text-xs font-bold text-gray-400 uppercase tracking-wider mb-2">Status / Coluna</label>
                            <select 
                              className="w-full bg-black/50 border border-white/10 rounded-xl px-4 py-3 text-white outline-none focus:border-blue-500/50"
                              value={editingTask.status} onChange={e => setEditingTask({...editingTask, status: e.target.value})}
                            >
                              <option value="inbox">Inbox Juan</option>
                              <option value="todo">Fila (A Fazer)</option>

                              <option value="review">Em Aprovação</option>
                            </select>

                      <div className="flex gap-3 mt-6 pt-4 border-t border-white/10">
                          <button type="button" onClick={() => moverParaFinalizado(editingTask.id)} className="flex-1 bg-emerald-600/20 hover:bg-emerald-500/30 text-emerald-400 border border-emerald-500/30 font-bold py-3 rounded-xl transition-colors">
                            <i className="fa-solid fa-check-double mr-2"></i> Finalizar
                          </button>
                          <button type="submit" className="flex-1 bg-blue-600 hover:bg-blue-500 text-white font-bold py-3 rounded-xl transition-colors">
                            Salvar Alterações
                          </button>
                      </div>
                    </form>
                  </div>
                </div>
              )}

            </div>
          );
        }

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
